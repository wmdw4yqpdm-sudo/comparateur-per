/* 🫧 BULLES FIX + RÉACTION SOURIS */ const canvas=document.getElementById("bubbleCanvas"); const ctx=canvas.getContext("2d"); let w=window.innerWidth; let h=window.innerHeight; function resize(){ w=window.innerWidth; h=window.innerHeight; canvas.width=w; canvas.height=h; } resize(); window.addEventListener("resize",resize); let mouse = { x: w/2, y: h/2 }; window.addEventListener("mousemove",(e)=>{ mouse.x = e.clientX; mouse.y = e.clientY; }); let bubbles=[]; function createBubbles(){ bubbles=[]; for(let i=0;i<80;i++){ bubbles.push({ x:Math.random()*w, y:Math.random()*h, r:Math.random()*5+2, vy:Math.random()*0.5+0.2 }); } } createBubbles(); window.addEventListener("resize",createBubbles); const colors=["#2563eb","#06b6d4","#22c55e","#94a3b8"]; function animate(){ ctx.clearRect(0,0,w,h); for(let b of bubbles){ /* mouvement vertical */ b.y -= b.vy; if(b.y < 0){ b.y = h; b.x = Math.random()*w; } /* 💫 effet souris */ let dx = (mouse.x - b.x) * 0.003; let dy = (mouse.y - b.y) * 0.003; b.x += dx; b.y += dy; ctx.beginPath(); ctx.fillStyle=colors[Math.floor(b.x % colors.length)]; ctx.arc(b.x,b.y,b.r,0,Math.PI*2); ctx.fill(); } requestAnimationFrame(animate); } animate();
